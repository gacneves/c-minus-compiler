int fibonacci(int n){
    int x; int y; int f; int i;
    x = 0;
    y = 1;
    i = 1;
    if(n == 0) return x;
    else{
        f = x + y;
        while(i<n){
            x = y;
            y = f;
            f = x + y;
            i = i + 1;
        }
        return f;
    }
}

void main(void){
    int n;
    n = input();
    output(fibonacci(n));
}
